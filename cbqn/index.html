cbqn

A crate for running BQN code within a Rust program using CBQN interpreter compiled as a shared object.

Usage

Simple expressions can be run with the BQN! convenience macro. For more advanced use, the methods of BQNValue provide the necessary functionality.

Examples using the BQN! macro

let sum = BQN!("1+1");
assert_eq!(sum.to_f64(), 2.0);
assert_eq!(BQN!("‚åΩ‚â°‚ä¢", "BQN").to_f64(), 0.0);
let strs = BQN!(' ', "(‚ä¢-Àú+`√ó¬¨)‚àò=‚äî‚ä¢", "Rust ‚ù§Ô∏è BQN")
    .to_bqnvalue_vec()
    .iter()
    .map(BQNValue::to_string)
    .collect::<Vec<String>>();
assert_eq!(strs, ["Rust", "‚ù§Ô∏è", "BQN"]);
let strings = ["join", "these", "please"];
assert_eq!(BQN!("‚àæ", strings).to_string(), "jointheseplease");

    .collect::<Vec<String>>();
